Preliminaries:

\begin{code}

data Σ {X : Set} (Y : X → Set) : Set where
 _,_ : (x : X)(y : Y x) → Σ {X} Y

π₀ : {X : Set} {Y : X → Set} → (Σ \(x : X) → Y x) → X
π₀(x , y) = x

π₁ : {X : Set} {Y : X → Set} (t : Σ \(x : X) → Y x) → Y(π₀ t)
π₁(x , y) = y

data ℕ : Set where
 zero : ℕ
 succ : ℕ → ℕ

{-# BUILTIN NATURAL ℕ #-}
{-# BUILTIN ZERO zero #-}
{-# BUILTIN SUC succ #-}

Baire : Set
Baire = ℕ → ℕ

data _≡_ {X : Set} : X → X → Set where
 refl : {x : X} → x ≡ x

ap : (X Y : Set) (f : X → Y) {x₀ x₁ : X} → x₀ ≡ x₁ → f x₀ ≡ f x₁
ap X Y f refl = refl

pred : ℕ → ℕ
pred 0 = 0
pred (succ n) = n

succ-injective : {m n : ℕ} → succ m ≡ succ n → m ≡ n
succ-injective = ap ℕ ℕ pred

succ-no-fix : (Σ \(m : ℕ) → m ≡ succ m) → 0 ≡ 1
succ-no-fix (0      , r) = r
succ-no-fix (succ m , r) = succ-no-fix (m , succ-injective r)

\end{code}

We consider r e = {e}, but, in fact, we don't need to know what r
really is. Any such r is impossible. So we just postulate any r:

\begin{code}

postulate r : ℕ → Baire

CT : Set
CT = (α : Baire) → Σ \(e : ℕ) → (n : ℕ) → r(e)(n) ≡ α n

CT-false : CT → 0 ≡ 1
CT-false ct = succ-no-fix (r n n , fixed-point-of-succ)
 where
  s : Baire → ℕ
  s α = π₀(ct α)

  CT' : (α : Baire) (n : ℕ) → r(s α)(n) ≡ α n
  CT' α = π₁(ct α)

  α : Baire
  α n = succ(r n n)

  n : ℕ
  n = s α

  fixed-point-of-succ : r n n ≡ succ(r n n)
  fixed-point-of-succ = CT' α n

\end{code}
