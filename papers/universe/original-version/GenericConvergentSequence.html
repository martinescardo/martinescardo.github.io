Martin Escardo 2012.

Sometimes I take the generic convergence sequence to be the set of
decreasing, infinite binary sequences. For the purposes of this
investigation, it is convenient to work with an isomorphic copy,
that of binary sequences that have at most one occurrence of
one. The constant sequence zero represents infinity. The advantage
is that a sequence represents the number n if and only if it has a
one at position n. Again, this with the relative Cantor topology
is (homeomorphic to) the one-point compactification of the natural
numbers. 

\begin{code}

{-# OPTIONS --without-K #-}

module GenericConvergentSequence where

open import Cantor
open import CurryHoward
open import Equality
open import Extensionality
open import Naturals
open import SetsAndFunctions
open import Two

has-at-most-one-1 : ₂ℕ → Prp
has-at-most-one-1 α = ∀(i j : ℕ) → α i ≡ ₁ → α j ≡ ₁ → i ≡ j

ℕ∞ : Set
ℕ∞ = Σ \(α : ₂ℕ) → has-at-most-one-1 α

incl : ℕ∞ → ₂ℕ
incl = π₀

_≣_ : ℕ∞ → ℕ → Prp
u ≣ n = incl u n ≡ ₁


\end{code}

We'll soon define an inclusion under : ℕ → ℕ∞ (for underline, as
underlining is the notation in papers), and show that u ≣ n is
equivalent to u ≡ under n. This means that it is decidable whether 
u = under n, or, in other words, that the element under n : ℕ∞ is
isolated. We'll soon define the non-isolated element ∞ (the negation
here is a figure of speach: what really happens is that the
isolatedness of ∞ amounts to WLPO, and so does the decidability of
equality of ℕ∞).

We first discuss the above inclusion incl : ℕ∞ → ₂ℕ of the subset ℕ∞
of the Cantor space ₂ℕ.

\begin{code}

incl-mono : ∀{u v : ℕ∞} →

  incl u ≡ incl v → u ≡ v

incl-mono {u} {v} = π₀-mono lemma u v
 where
  open import FirstProjectionInjective

  lemma : ∀(α : ₂ℕ) → ∀(p q : has-at-most-one-1 α) → p ≡ q
  lemma α p q = extensionality(λ i → extensionality(fact₂ i))
   where
    fact₀ : ∀{i j} → ∀(f g : α i ≡ ₁ → α j ≡ ₁ → i ≡ j) → f ≡ g
    fact₀ f g = extensionality(λ r → extensionality(fact₁ r))
     where
      fact₁ : ∀ r s → f r s ≡ g r s
      fact₁ r s = ℕ-uip (f r s) (g r s)

    fact₂ : ∀ i j → p i j ≡ q i j
    fact₂ i j = fact₀ (p i j) (q i j)

\end{code}

We now construct the elements of ℕ∞ and develop basic facts: 

\begin{code}

Zero : ℕ∞
Zero = (α , reason)
 where
  α : ₂ℕ
  α 0 = ₁
  α(succ n) = ₀
  reason : has-at-most-one-1 α
  reason 0 0 = λ r s → refl
  reason i (succ j) = λ r ()
  reason (succ i) j = λ()


Succ : ℕ∞ → ℕ∞
Succ (α , reason) = (α' , reason')
 where
  α' : ₂ℕ
  α' 0 = ₀
  α'(succ n) = α n
  reason' : has-at-most-one-1 α'
  reason' 0 0 = λ r s → refl
  reason' 0 (succ j) = λ()
  reason' (succ i) 0 = λ r ()
  reason' (succ i) (succ j) = λ r s → cong succ (reason i j r s)


under : ℕ → ℕ∞
under 0 = Zero
under (succ n) = Succ(under n)


∞ : ℕ∞
∞ = ((λ i → ₀) , λ i j ())


under-lemma₀ : ∀(m n : ℕ) → under m ≣ n → m ≡ n
under-lemma₀ 0 0 r = refl
under-lemma₀ 0 (succ n) r = ⊥-elim(zero-is-not-one r)
under-lemma₀ (succ m) 0 r = ⊥-elim(zero-is-not-one r)
under-lemma₀ (succ m) (succ n) r = cong succ(under-lemma₀ m n r)


under-lemma₁ : ∀(n : ℕ) → under n ≣ n
under-lemma₁ 0 = refl
under-lemma₁(succ n) = under-lemma₁ n


under-lemma₂ : ∀(u : ℕ∞) → ∀ n → u ≡ under n → u ≣ n
under-lemma₂ u n r = trans (cong (λ f → f n) (cong incl r)) (under-lemma₁ n)


under-lemma₃ : ∀(u : ℕ∞) → ∀ n → u ≣ n → u ≡ under n
under-lemma₃ u n r = incl-mono(extensionality lemma₂)
 where
  lemma₀ : ∀(i : ℕ) → n ≡ i → incl u i ≡ incl (under n) i
  lemma₀ i s = subst s (Lemma[x≡z→y≡z→x≡y] r (under-lemma₁ n))

  lemma₁ : ∀(i : ℕ) → n ≠ i → incl u i ≡ incl (under n) i
  lemma₁ i f = Lemma[x≡z→y≡z→x≡y] fact₀ fact₁
   where
    fact₀ : incl u i ≡ ₀
    fact₀ = Lemma[b≠₁→b≡₀](contra-positive(π₁ u n i r) f)
    fact₁ : incl (under n) i ≡ ₀
    fact₁ = Lemma[b≠₁→b≡₀](contra-positive(under-lemma₀ n i) f)

  lemma₂ : ∀(i : ℕ) → incl u i ≡ incl (under n) i
  lemma₂ i = ∨-elim (lemma₀ i) (lemma₁ i) (ℕ-discrete n i)


not-ℕ-is-∞ : ∀(u : ℕ∞) → (∀(n : ℕ) → u ≠ under n) → u ≡ ∞
not-ℕ-is-∞ u f = incl-mono(extensionality lemma)
 where
  lemma : ∀(n : ℕ) →  incl u n ≡ incl ∞ n
  lemma n = Lemma[b≠₁→b≡₀](contra-positive(under-lemma₃ u n)(f n))

\end{code}
