cross :: Integer -> [Maybe a] -> [Maybe a]
cross n = h (n-1)
where
h 0 (p : s) = Nothing : h (n-1) s
h i (p : s) = p : h (i-1) s

sieve :: [Maybe Integer] -> [Maybe Integer]
sieve (Nothing : s) = Nothing : sieve s
sieve (Just p : s) = Just p : sieve (cross p s)
