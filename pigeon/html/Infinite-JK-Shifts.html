-- In the following use either J-∀-shift-bbc or else J-∀-shift-selection:

J-∀-shift : {R : Ω} {A : ℕ → Ω} →
---------

      (∀(n : ℕ) → J {R} (A n)) → J {R} (∀(n : ℕ) → A n)

J-∀-shift = J-∀-shift-selection


-- The remainder of this module can be left untouched.

---------------------------------------------------------
-- We reduce the double negation shift to the J-shift. --
---------------------------------------------------------

-- We need a hypothesis, which amounts to ex falso quodlibet, with R
-- playing the role of false.


K-∀-shift : {R : Ω} {A : ℕ → Ω} →
---------
            (∀(n : ℕ) → R → A n) →                   -- efqs,
            (∀(n : ℕ) → K(A n)) → K(∀(n : ℕ) → A n)  -- shift.

K-∀-shift efqs φs = J-K(J-∀-shift(λ n → K-J(efqs n) (φs n)))
