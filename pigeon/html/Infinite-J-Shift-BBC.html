{-# OPTIONS --no-termination-check #-}

-- This is an alternative to Infinite-JK-Shifts-Selection.agda. We use
-- the Berardi-Bezem-Coquand functional to realize the J-Shift and
-- hence the K-Shift.

module Infinite-J-Shift-BBC where

open import Logic
open import LogicalFacts
open import Naturals
open import JK-Monads


J-∀-shift-bbc : {R : Ω} {A : ℕ → Ω} →
-------------

      (∀(n : ℕ) → J(A n)) → J(∀(n : ℕ) → A n)

J-∀-shift-bbc {R} {A} ε p = bbc p
  where
   bbc : J {R} (∀(n : ℕ) → A n)
   bbc p i = ε i (λ x → J-K bbc (p ∘ mapsto {A} i x))
