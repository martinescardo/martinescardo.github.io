-- We prove a theorem that involves the classical existential
-- quantifier K∃ (called pigeonhole below). The proof uses excluded
-- middle and classical countable choice (i.e. choice formulated with
-- the classical existential quantifier), which is implemented using
-- the K-shift (more commonly known as the double negation shift) in
-- the modules JK-Choice.agda and Infinite-JK-Shifts.agda.
--
-- In the module FinitePigeon.agda we derive a statement that uses the
-- intuitionistic quantifiers (and doesn't mention the double negation
-- modality K at all), using the classical result as a lemma. In the
-- module Examples.agda we run it.


-- Definition 

Pigeonhole : {R : Ω} → ₂ℕ → Ω
Pigeonhole {R} α = ∃ \(b : ₂) → ∃ \(g : ℕ → ℕ) → ∀(i : ℕ) → g i < g(i + 1) ∧ K {R} (α(g i) ≡ b)


pigeonhole : {R : Ω} →
----------

    ∀(α : ₂ℕ) → K(Pigeonhole α)

pigeonhole {R} α = lemma₈
 where
  A : Ω
  A = K∃ \(n : ℕ) → ∀(i : ℕ) → K(α(n + i) ≡ ₀)

  case₁ : A → K(Pigeonhole α)
  case₁ a = K-functor lemma₁ a
   where
    lemma₁ : (∃ \(n : ℕ) → ∀(i : ℕ) → K(α(n + i) ≡ ₀)) → Pigeonhole α
    lemma₁ (∃-intro n h) = ∃-intro ₀ (∃-intro (λ i → n + i) λ i → (∧-intro ( ∃-intro 0 reflexivity) (h i)))

  case₂ : (A → R) → K(Pigeonhole α)
  case₂ assumption =  K-functor lemma₇ lemma₆
   where
    lemma₂ : ∀(n : ℕ) → (∀(i : ℕ) → K(α(n + i) ≡ ₀)) → R
    lemma₂ = not-exists-implies-forall-not(three-negations-imply-one assumption)

    lemma₃ : ∀(n : ℕ) → K∃ \(i : ℕ) → K(α(n + i) ≡ ₁)
    lemma₃ = lemma₄ lemma₂
     where lemma₄ : (∀(n : ℕ) → (∀(i : ℕ) → K(α(n + i) ≡ ₀)) → R) →
               (∀(n : ℕ) → K∃ \(i : ℕ) → K(α(n + i) ≡ ₁))
           lemma₄ h n = K-functor lemma₅ (not-forall-not-implies-K-exists(h n))
            where lemma₅ : (∃ \(i : ℕ) → α(n + i) ≡ ₀ → R) → ∃ \(i : ℕ) → K(α(n + i) ≡ ₁)
                  lemma₅ (∃-intro i r) = (∃-intro i (two-equality-cases (α(n + i)) r))

    lemma₆ : K∃ \(f : ℕ → ℕ) → ∀(n : ℕ) → K(α(n + 1 + f n) ≡ ₁)
    lemma₆ = K-AC-ℕ (λ n → λ r → ∃-intro 0 (λ p → r)) (λ n → lemma₃(n + 1))

    lemma₇ : (∃ \(f : ℕ → ℕ) → ∀(n : ℕ) → K(α(n + 1 + f n) ≡ ₁)) → Pigeonhole α
    lemma₇ (∃-intro f h) = ∃-intro ₁ (∃-intro g λ i → (∧-intro (fact₀ i) (fact₁ i)))
     where g : ℕ → ℕ
           g 0 = 0 + 1 + f 0
           g(succ i) = let i' = g i in  i' + 1 + f i'

           fact₀ : ∀(i : ℕ) → g i < g i + 1 + f(g i)
           fact₀ i = ∃-intro (f(g i)) (fact (g i) (f(g i)))
            where fact : ∀(x y : ℕ) → x + y + 1 ≡ x + 1 + y
                  fact x y = transitivity (addition-associativity x y 1)
                            (transitivity (compositionality (λ z → x + z) (addition-commutativity y 1))
                            (symmetry (addition-associativity x 1 y)))

           fact₁ : ∀(i : ℕ) → K(α(g i) ≡ ₁)
           fact₁ 0 = h 0
           fact₁ (succ i) = h(g i)

  lemma₈ : K(Pigeonhole α)
  lemma₈ = K-∨-elim case₁ case₂ K-Excluded-Middle
