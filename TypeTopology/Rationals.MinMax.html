max' : (x y : ‚Ñö) ‚Üí (x < y) ‚àî (x Ôºù y) ‚àî (y < x) ‚Üí ‚Ñö
max' x y (inl _) = y
max' x y (inr _) = x

max : ‚Ñö ‚Üí ‚Ñö ‚Üí ‚Ñö
max p q = max' p q (‚Ñö-trichotomous p q)

max'-to-max : (x y : ‚Ñö)
            ‚Üí (t : (x < y) ‚àî (x Ôºù y) ‚àî (y < x))
            ‚Üí max' x y t Ôºù max x y
max'-to-max x y t = equality-cases t I II
 where
  I : (l : x < y) ‚Üí t Ôºù inl l ‚Üí max' x y t Ôºù max x y
  I l e = max' x y t                    Ôºù‚ü® ap (max' x y) e                    ‚ü©
          max' x y (inl l)              Ôºù‚ü® ap (max' x y) (‚Ñö-trich-a x y l ‚Åª¬π) ‚ü©
          max' x y (‚Ñö-trichotomous x y) Ôºù‚ü® by-definition                      ‚ü©
          max x y                       ‚àé

  II : (l : (x Ôºù y) ‚àî y < x) ‚Üí t Ôºù inr l ‚Üí max' x y t Ôºù max x y
  II r e = max' x y t                    Ôºù‚ü® ap (max' x y) e                    ‚ü©
           max' x y (inr r)              Ôºù‚ü® ap (max' x y) (‚Ñö-trich-b x y r ‚Åª¬π) ‚ü©
           max' x y (‚Ñö-trichotomous x y) Ôºù‚ü® by-definition                      ‚ü©
           max x y                       ‚àé

max'-refl : (q : ‚Ñö) ‚Üí (t : (q < q) ‚àî (q Ôºù q) ‚àî (q < q)) ‚Üí max' q q t Ôºù q
max'-refl q (inl l) = ùüò-elim (‚Ñö<-irrefl q l)
max'-refl q (inr (inl l)) = l
max'-refl q (inr (inr l)) = ùüò-elim (‚Ñö<-irrefl q l)

max-refl : (q : ‚Ñö) ‚Üí max q q Ôºù q
max-refl q = I (‚Ñö-trichotomous q q)
 where
  I : (q < q) ‚àî (q Ôºù q) ‚àî (q < q) ‚Üí max q q Ôºù q
  I t = max q q    Ôºù‚ü® max'-to-max q q t ‚Åª¬π ‚ü©
        max' q q t Ôºù‚ü® max'-refl q t        ‚ü©
        q          ‚àé

max'-comm : (x y : ‚Ñö)
            ‚Üí (s : (x < y) ‚àî (x Ôºù y) ‚àî (y < x))
            ‚Üí (t : (y < x) ‚àî (y Ôºù x) ‚àî (x < y))
            ‚Üí max' x y s Ôºù max' y x t
max'-comm x y (inl s) (inl t)             = ùüò-elim (‚Ñö<-irrefl x (‚Ñö<-trans x y x s t))
max'-comm x y (inl s) (inr (inl t))       = ùüò-elim (‚Ñö<-irrefl y (transport (_< y) (t ‚Åª¬π) s))
max'-comm x y (inl s) (inr (inr t))       = refl
max'-comm x y (inr (inl s)) (inl t)       = refl
max'-comm x y (inr (inr s)) (inl t)       = refl
max'-comm x y (inr (inl s)) (inr (inl t)) = s
max'-comm x y (inr (inl s)) (inr (inr t)) = s
max'-comm x y (inr (inr s)) (inr (inl t)) = t ‚Åª¬π
max'-comm x y (inr (inr s)) (inr (inr t)) = ùüò-elim (‚Ñö<-irrefl x (‚Ñö<-trans x y x t s))

max-comm : (p q : ‚Ñö) ‚Üí max p q Ôºù max q p
max-comm x y =
 max x y                       Ôºù‚ü® max'-to-max x y (‚Ñö-trichotomous x y)                    ‚ü©
 max' x y (‚Ñö-trichotomous x y) Ôºù‚ü® max'-comm x y (‚Ñö-trichotomous x y) (‚Ñö-trichotomous y x) ‚ü©
 max' y x (‚Ñö-trichotomous y x) Ôºù‚ü® max'-to-max y x (‚Ñö-trichotomous y x)                    ‚ü©
 max y x                       ‚àé

max'-to-‚â§ : (p q : ‚Ñö) ‚Üí (t : (p < q) ‚àî (p Ôºù q) ‚àî (q < p))
                      ‚Üí (p ‚â§ q) √ó (max' p q t Ôºù q)
                      ‚àî (q ‚â§ p) √ó (max' p q t Ôºù p)
max'-to-‚â§ p q (inl t) = Cases (‚Ñö-trichotomous p q) I II
 where
  I : p < q ‚Üí (p ‚â§ q) √ó (max' p q (inl t) Ôºù q) ‚àî q ‚â§ p √ó (max' p q (inl t) Ôºù p)
  I l = inl ((‚Ñö<-coarser-than-‚â§ p q l) , refl)

  II : (p Ôºù q) ‚àî q < p
     ‚Üí (p ‚â§ q)
     √ó (max' p q (inl t) Ôºù q) ‚àî q ‚â§ p √ó (max' p q (inl t) Ôºù p)
  II (inl e) = ùüò-elim (‚Ñö<-irrefl p (transport (p <_) (e ‚Åª¬π) t))
  II (inr l) = ùüò-elim (‚Ñö<-irrefl p (‚Ñö<-trans p q p t l))
max'-to-‚â§ p q (inr t) = inr (I t , refl)
 where
  I : (p Ôºù q) ‚àî q < p ‚Üí q ‚â§ p
  I (inl l) = transport (q ‚â§_) (l ‚Åª¬π) (‚Ñö‚â§-refl q)
  I (inr l) = ‚Ñö<-coarser-than-‚â§ q p l

max-to-‚â§ : (p q : ‚Ñö) ‚Üí (p ‚â§ q) √ó (max p q Ôºù q) ‚àî q ‚â§ p √ó (max p q Ôºù p)
max-to-‚â§ p q = I (max'-to-‚â§ p q (‚Ñö-trichotomous p q))
 where
  I : (p ‚â§ q) √ó (max' p q (‚Ñö-trichotomous p q) Ôºù q)
    ‚àî (q ‚â§ p) √ó (max' p q (‚Ñö-trichotomous p q) Ôºù p)
    ‚Üí (p ‚â§ q) √ó (max p q Ôºù q) ‚àî (q ‚â§ p) √ó (max p q Ôºù p)
  I (inl t) = inl t
  I (inr t) = inr t

decide-max : (p q : ‚Ñö) ‚Üí (max p q Ôºù q) ‚àî (max p q Ôºù p)
decide-max p q = +functor pr‚ÇÇ pr‚ÇÇ (max-to-‚â§ p q)

max‚â§ : (p q : ‚Ñö) ‚Üí p ‚â§ max p q
max‚â§ p q = I (max-to-‚â§ p q)
 where
  I : (p ‚â§ q) √ó (max p q Ôºù q) ‚àî (q ‚â§ p) √ó (max p q Ôºù p) ‚Üí p ‚â§ max p q
  I (inl (p‚â§q , e)) = transport (p ‚â§_) (e ‚Åª¬π) p‚â§q
  I (inr (q‚â§p , e)) = transport (p ‚â§_) (e ‚Åª¬π) (‚Ñö‚â§-refl p)

max‚â§' : (p q : ‚Ñö) ‚Üí q ‚â§ max p q
max‚â§' p q = transport (q ‚â§_) (max-comm q p) (max‚â§ q p)

min' : (x y : ‚Ñö) ‚Üí (x < y) ‚àî (x Ôºù y) ‚àî (y < x) ‚Üí ‚Ñö
min' x y (inl _) = x

min : ‚Ñö ‚Üí ‚Ñö ‚Üí ‚Ñö

min'-to-min : (x y : ‚Ñö)

  I : (k : x < y) ‚Üí t Ôºù inl k ‚Üí min' x y t Ôºù min x y

          min' x y (inl k)               Ôºù‚ü® ap (min' x y) (‚Ñö-trich-a x y k ‚Åª¬π) ‚ü©
          min' x y (‚Ñö-trichotomous x y)  Ôºù‚ü® by-definition                      ‚ü©
          min x y                        ‚àé

  II : (k : (x Ôºù y) ‚àî y < x) ‚Üí t Ôºù inr k ‚Üí min' x y t Ôºù min x y
  II k e = min' x y t                    Ôºù‚ü® ap (min' x y) e                    ‚ü©
           min' x y (inr k)              Ôºù‚ü® ap (min' x y) (‚Ñö-trich-b x y k ‚Åª¬π) ‚ü©
           min' x y (‚Ñö-trichotomous x y) Ôºù‚ü® by-definition                      ‚ü©

            ‚Üí (s : (x < y) ‚àî (x Ôºù y) ‚àî (y < x))
